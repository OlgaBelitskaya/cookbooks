$$\mathbb{3D \; Plotting \; of \; Functions}$$


@interact
def _(a=[3,4,5,6],b=[3,4,5,6]):
    var('x,y,z'); c=colormaps.jet(8*a*b)[:3]
    f=a*(cos(x)+cos(y)+cos(z))+b*cos(x)*cos(y)*cos(z)
    p=implicit_plot3d(f,(x,-b,b),(y,-b,b),(z,-b,b),
                      mesh=True,plot_points=10,
                      alpha=.5,color=c)
    p.show(frame=False)
