$\large{\mathbb{Interactive \; 3D \; Plotting}}$

Click the $\large{\mathbb{run}}$ button to display an example.


@interact
def _(gnum=[3,..,6],lnum=[2,..,5]):
    LG=Graphics(); g=graphs.CubeGraph(gnum)
    vg=g.get_pos(); eg=g.edges() 
    nv=len(vg); ne=len(eg); k=randint(10,50)/100
    x0=mean([list(vg.values())[i][0] for i in range(nv)])
    y0=mean([list(vg.values())[i][1] for i in range(nv)])
    coord=[[list(vg[eg[i][0]]),list(vg[eg[i][1]])] 
           for i in range(ne)]
    lg=[[[x-x0,y-y0] for [x,y] in coord[i]] 
        for i in range(ne)]
    zlg=[[[(1+k*j)*x,j,(1+k*j)*y] 
          for [x,y] in lg[i]] 
         for j in range(lnum) for i in range(ne)]
    for i in range(lnum*ne): 
        LG+=line3d(zlg[i],
                   color=(floor(i/ne)/(lnum-1),
                          random(),random()))
    LG.show(frame=False,aspect_ratio=1)
