$$\mathbb{ Set \; Operations }$$


@interact
def _(A='a,3,2,c,1,ф,⎈,D',
      B='ф,7,a,⎈,2,1,B,θ',
      C='8,N,a,θ,4,P,⎈,r'):
    S=[sorted(list(el.split(','))) 
       for el in [A,B,C]]     
    st='<style>h4 {color:#ff3636; font-size:20px}'+\
       'h5 {color:#3636ff; font-size:20px}</style>'     
    s1='<h5>$A \\cap B = %s$</h5>'
    s2='<h5>$A \\cap C = %s$</h5>'
    s3='<h5>$B \\cap C = %s$</h5>'
    s4='<h4>$A \\cap B \\cap C = %s$</h5>'
    AB=list(set(S[0])&set(S[1]))
    AC=list(set(S[0])&set(S[2]))
    BC=list(set(S[1])&set(S[2])) 
    ABC=list(set(AB)&set(S[2]))
    pretty_print(html(st+s1%('{'+','.join(AB)+'}')))
    pretty_print(html(st+s2%('{'+','.join(AC)+'}')))
    pretty_print(html(st+s3%('{'+','.join(BC)+'}')))
    pretty_print(html(st+s4%('{'+','.join(ABC)+'}')))
