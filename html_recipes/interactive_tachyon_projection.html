Click the $\large{\mathbb{run}}$ button to display an example.


@interact
def _(projection=['perspective','fisheye','perspective_dof'],
      cmap=['hsv','jet','cool','hot','prism']):
    k=1; e=[[[k,-k,k],[0,0,0]],[[0,0,0],[k,k,k]],
            [[k,k,k],[-k,k,k]],[[-k,k,k],[-k,-k,k]],
            [[-k,-k,k],[k,-k,k]],[[k,-k,k],[k,k,k]],
            [[k,k,k],[k,k,-k]],[[k,k,-k],[-k,k,-k]],
            [[-k,k,-k],[-k,-k,-k]],[[-k,-k,-k],[k,-k,-k]],
            [[k,-k,-k],[k,k,-k]],[[k,k,-k],[0,0,0]],
            [[0,0,0],[k,-k,-k]],[[k,-k,-k],[k,-k,k]]]
    if projection=='perspective_dof':
        t=Tachyon(xres=450,yres=450,projection=projection,
                  camera_center=(-3.5,0.,0.),
                  antialiasing=3,raydepth=5,
                  focallength='2.0',aperture='.003')
    else: t=Tachyon(xres=450,yres=450,projection=projection,
                    camera_center=(-3.5,0.,0.))
    for i in [0..13]:
        t.texture('t%s'%str(i),opacity=.3,
                  color=colormaps[cmap](.077*i)[:3])
        t.fcylinder(e[i][0],e[i][1],.06,'t%s'%str(i)) 
    t.light((-5,-5,5),.1,(1,1,1)); t.show()
