 c=list(colors.keys()); f(x,t)=sin(5*x)^2+0.02*t 
 @interact
 def _(number=[20,28,..,len(c)]):
     p=sum([polar_plot(f(x,i),(x,0,2*pi),plot_points=30,
                       thickness=1,color=c[i-1]) 
            for i in [1..number]])
     p.show(figsize=(5,5),gridlines=True)
